dan ini si javascriptnya :

{% highlight javascript %} var app = angular.module('App', []).controller('MultiSelectCtrl', ['$scope', function($scope) { $('.selected-items-box').bind('click', function(e) { $('.wrapper .list').slideToggle('fast'); }); $scope.friends = [ {name:'Lucky'}, {name:'Yoga'}, {name:'Fauzan'}, {name:'Cecep'}, {name:'Nurjaman'}, {name:'Asti'} ]; $scope.getSelectedItems = function(item){ return item.selected; }; } ]); {% endhighlight %}
